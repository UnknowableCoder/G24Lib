632  template <g24_lib::Definitions::default_unsigned_indexer order = 0>
633  CUDA_HOS_DEV static auto evolve( Func_T &&f, Up_T &&update, object && obj, const time_type &current,
634  const time_type &dt, const tol_type &tol,
635  storage &store, Args&&... params)
636  {
637  if constexpr(butcher_tableau::iterations_needed == 1)
638  {
639  f(store.temp[0], std::forward<object>(obj), current, dt, std::forward<Args>(params)...);
640  auto&& final_1 = final_iteration<typename butcher_tableau::result>(obj, store, dt);
641  //No forwarding here because we deal with const references.
642  return update(std::forward<object>(obj), std::forward<decltype(final_1)>(final_1),
643  dt, tol, std::forward<Args>(params)...);
644  }
645  else if constexpr(order == 0)
646  {
647  f(store.temp[0], std::forward<object>(obj), current, dt, std::forward<Args>(params)...);
648  return evolve<1>(std::forward<Func_T>(f), std::forward<Up_T>(update),
649  std::forward<object>(obj), current, dt, tol, store, std::forward<Args>(params)...);
650  }
651  else if constexpr(order == butcher_tableau::iterations_needed - 1)
652  {
653  using entry = typename butcher_tableau::template entry<order>;
654 
655  auto&& value = value_iteration<entry>(obj, store, dt);
656 
657  f(store.temp[order], std::forward<decltype(value)>(value), current + entry::evaluation_time*dt,
658  dt, std::forward<Args>(params)...);
659 
660 
661  if constexpr (butcher_tableau::is_extended)
662  {
663 
664  auto&& final_1 = final_iteration<typename butcher_tableau::first_result>(obj, store, dt);
665 
666  auto&& final_2 = final_iteration<typename butcher_tableau::second_result>(obj, store, dt);
667 
668  return update(std::forward<object>(obj), std::forward<decltype(final_2)>(final_2),
669  std::forward<decltype(final_1)>(final_1), dt, tol, std::forward<Args>(params)...);
670  }
671  else
672  {
673  auto&& final_1 = final_iteration<typename butcher_tableau::result>(obj, store, dt);
674 
675  return update(std::forward<object>(obj), std::forward<decltype(final_1)>(final_1),
676  nullptr, dt, tol, std::forward<Args>(params)...);
677  }
678  }
679  else
680  {
681  using entry = typename butcher_tableau::template entry<order>;
682 
683  auto value = value_iteration<entry>(obj, store, dt);
684 
685  f(store.temp[order], value, current + entry::evaluation_time*dt, dt, params...);

986  template <class object, class time_type, class tol_type, class NewT1, class NewT2, class ... Args>
987  CUDA_HOS_DEV inline auto trivial_updater( object &obj, const NewT1 &recent1,
988  const NewT2 &recent2, const time_type &dt, const tol_type &tolerance,
989  Args&& ... params )
990  {
991  using namespace std;
992  //For abs() and CUDA interoperatability.
993  using err_diff_T = decltype(abs(std::declval<object>()));
994  if constexpr (std::is_null_pointer_v<NewT2>)
995  //Non-adaptive method
996  {
997  simple_evolve_result<err_diff_T, err_diff_T,time_type> ret;
998  ret.dt_used = dt;
999  ret.next_dt = dt;
1000  ret.error = std::numeric_limits<err_diff_T>::max();
1001  ret.difference = abs(-obj + recent1);
1002  obj = object(recent1);
1003  return ret;
1004  }
1005  else
1006  //Adaptive method
1007  {
1008  simple_evolve_result<err_diff_T, err_diff_T,time_type> ret;
1009  ret.dt_used = dt;
1010  ret.error = abs(abs(recent1)-abs(recent2));
1011  ret.next_dt = time_type(dt * pow(tolerance/(2*(ret.error)),0.25));
1012  if (ret.next_dt < 0.01 * dt)
1013  {
1014  ret.next_dt = 0.01 * dt;
1015  }
1016  else if (ret.next_dt > 100 * dt)
1017  {
1018  ret.next_dt = 100 * dt;
1019  }
1020  ret.difference = abs(-obj + recent1);
1021  obj = object(recent1);
1022  return ret;
1023  }
1024  }
