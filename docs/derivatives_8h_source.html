116  template <class Tarr> CUDA_HOS_DEV inline static auto derivative(const Tarr &arr,
117  const indexer elem,
118  const indexer offset = 0,
119  const indexer sep = 1)
120  {
121  if constexpr (!fin_diff_coeff::valid())
122  {
123  exceptions::throw_exception(exceptions::undefined_derivative("Undefined (as of now) derivative [1]."));
124  return arr[elem];
125  }

149  template <class Tarr> CUDA_HOS_DEV inline static
150  g24_lib::value_type<Tarr> derivate(const Tarr &arr,
151  const indexer elem,
152  const indexer sep = 1,
153  const indexer left_constraint = std::numeric_limits<indexer>::max()/2 - 1,
154  const indexer right_constraint= std::numeric_limits<indexer>::max()/2 )
155  {
156  if constexpr (npoints <= 1 || !fin_diff_coeff::valid())
157  {
158  exceptions::throw_exception(exceptions::undefined_derivative("Undefined (as of now) derivative [2]."));
159  return arr[elem];
160  }
161  else
162  {
163  indexer max = left_constraint + right_constraint + 1;
164  if(max < npoints)
165  {
166  return derivator<indexer, factorT, ord, npoints-1>::derivate(arr, elem, sep, left_constraint, right_constraint);
167  }
168  else if(max > npoints && npoints % 2 == 0)
169  {
170  return derivator<indexer, factorT, ord, npoints+1>::derivate(arr, elem, sep, left_constraint, right_constraint);
171  //So we typically use centered derivatives whenever possible,
172  //for greater numerical performance without much additional cost.
173  }
174  else if (npoints % 2 != 0)
175  {
176  //REMEMBER: npoints/2 is rounded down,
177  //so npoints = npoints/2 + npoints/2 + 1
178  //since we should be dealing with odd cases.
179  indexer left_side = std::min(left_constraint, npoints/2) + npoints/2-std::min(right_constraint, npoints/2);
180  //Since max < npoints already takes care of the cases
181  //where left_constraint < npoints/2 && right_constraint < npoints/2,
182  //if left_constraint < npoints/2,
183  //right_constraint > npoints/2,
184  //which means we get std::min(left_constraint, npoints/2) == left_constraint
185  //and std::min(right_constraint, npoints/2) == npoints/2
186  //and left_side = left_constraint + npoints/2 - npoints/2 == left_constraint,
187  //while right_constraint < npoints/2 implies
188  //left_constraint > npoints/2,
189  //yielding std::min(left_constraint, npoints/2) == npoints/2
190  //and std::min(right_constraitn, npoints/2) == npoints/2
191  //and left_side = npoints/2 + npoints/2 - right_constraint == npoints - 1 - right_constraint.
192  //This is exactly the behaviour we wanted.
193  return derivative(arr, elem, left_side, sep);
194  }
195  else
196  //npoints == max and even npoints,
197  //no choice without dropping accuracy.
198  {
199  return derivative(arr, elem, left_constraint, sep);
200  }
201  }
202  }
